styling/theme_loader.py
import json
from pathlib import Path

class ThemeLoader:
    def __init__(self, theme_name):
        theme_path = Path(__file__).parent / "themes" / f"{theme_name.lower().replace(' ', '_')}.json"
        with open(theme_path, "r") as f:
            self.theme_data = json.load(f)

    def get(self, key, default=None):
        return self.theme_data.get(key, default)

    def apply_to_widget(self, widget, element):
        """Apply style dynamically to a widget"""
        data = self.theme_data.get(element, {})
        if isinstance(data, dict):
            bg = data.get("color", "")
            text = data.get("text", "")
            hover = data.get("hover", "")
            widget.setStyleSheet(f"""
                QPushButton {{
                    background-color: {bg};
                    color: {text};
                }}
                QPushButton:hover {{
                    background-color: {hover};
                }}
            """)
        else:
            widget.setStyleSheet(f"background-color: {data};")

from PySide6.QtWidgets import QApplication, QComboBox, QPushButton, QWidget, QVBoxLayout
from styling.theme_loader import ThemeLoader

class MainWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.dropdown = QComboBox()
        self.dropdown.addItems(["Ocean Sunset", "Ocean Breeze", "Blush Romance"])
        self.dropdown.currentTextChanged.connect(self.switch_theme)

        self.main_btn = QPushButton("Main Button")
        self.branch_btn = QPushButton("Branch 1")

        layout = QVBoxLayout(self)
        layout.addWidget(self.dropdown)
        layout.addWidget(self.main_btn)
        layout.addWidget(self.branch_btn)

        self.switch_theme("Ocean Sunset")  # Default

    def switch_theme(self, theme_name):
        self.theme = ThemeLoader(theme_name)
        self.setStyleSheet(f"background-color: {self.theme.get('background')};")
        self.theme.apply_to_widget(self.main_btn, "main_button")
        self.theme.apply_to_widget(self.branch_btn, "branch_button1")

app = QApplication([])
w = MainWindow()
w.show()
app.exec()
